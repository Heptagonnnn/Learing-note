## dependencies

```javascript
// 表示包依赖关系

// 版本号规则
// 主号.次号.修补号

"dependencies": {
  // 表示指定版本号
  "aaa": "1.1.1",
  // > 表示大于该版本号
  "bbb": ">1.1.1",
  // >= 表示大于等于该版本号
  "ccc": ">=1.1.1",
  // < 表示小于该版本号
  "ddd": "<1.1.1",
  // <= 表示小于等于该版本号
  "eee": "<=1.1.1",
  // ~
  // 表示版本号>=1.1.1，修补号>=1的任意值
  "fff": "~1.1.1",
  // 表示版本号>=1.1，修补号>=0的任意值
  "ggg": "~1.1",
  // 表示此版本号任意，修补号任意
  "hhh": "~1",

  // ^
  // 表示从左向右，第一个非0右侧任意
  // 主版本号固定，次版本号固定，修补号>=2的任意值
  "iii": "^0.1.2",
  // 主版本号固定，次版本号>=2，修补号任意，但总满足版本号大于1.2.3
  "jjj": "^1.2.3",
  // 版本号>=0.1，缺少的版本号位置为任意值
  "lll": "^0.1",

  // 空或*表示版本任意(没见过)
  "mmm": "",
  "nnn": "*",

  // - 表示版本号范围
  "ooo" "1.1.1-1.2.5"
}
```
